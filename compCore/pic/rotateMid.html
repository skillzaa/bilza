import Bilza, { Ui,hsl} from "../../src/bilza.js";

let bil = new Bilza("bilza", 70);
bil.background.color.set(hsl(240,100,30));

const grid = bil.alwaysOn().grid(10,10,"grey");

const fc = bil.alwaysOn().frameCounter(0);
fc.align(0,2);
fc.goto(0,0,100);
fc.fontSize.set(80);
fc.color.set(hsl(60));
fc.colorBackground.set(hsl(240,100,60));

const comp = bil.add(0,20).pic("../../images/cat.jpg");
comp.align(1,1);
comp.alignRotate(1,1);
comp.setxy(50,50);
comp.setPaddings(10);

comp.width.set(40)
comp.height.set(30)

comp.showBackground.set(true);
comp.colorBackground.set(hsl(60,100,90));
comp.border.set(2);
comp.colorBorder.set(hsl(0));
comp.rotation.animate(0,15, 0, 3 * 360);

const ui = new Ui(bil);
bil.draw();
