import Bilza, { Ui,hsl} from "../../src/bilza.js";

let bil = new Bilza("bilza", 70);

const grid = bil.alwaysOn( ).grid(10,10,"grey");

const comp = bil.add(0,65).particleSystem(10,hsl(0),2);
comp.particleSize.set(3);
comp.align(1,1);
comp.alignRotate(1,1);

comp.showBackground.set(true);
comp.colorBackground.set(hsl(120,20));

comp.setPaddings(10);

comp.border.set(4);
comp.colorBorder.set(hsl(0));

comp.x.set(50);
comp.y.set(50);

comp.width.set(10);
comp.height.set(10);

comp.width.oscillate(1,60,10,60,5);
comp.height.oscillate(1,60,10,60,5);

const ui = new Ui(bil);
bil.draw();
