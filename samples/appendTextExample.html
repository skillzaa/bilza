 function getTxt(content){
let txt = bil.add.text(5,content,"#008000",10,25,80,50);
txt.insertType = txt.insertTypeOptions.Append;
txt.xAlignment = txt.xAlignmentOptions.Mid; 
txt.yAlignment = txt.yAlignmentOptions.Mid; 
bil.insert(txt);
return txt;    
 }
