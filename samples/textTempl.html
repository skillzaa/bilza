import Bilza, { Ui, CompFactory as cf, hsl,TextTempl } from "../bilza/src/bilza.js";

const bil = new Bilza ("bilza",80);
bil.resizeCanvas(600,300);

bil.insert.alwaysOn(cf.frameCounter(hsl(180)));

const txt = TextTempl.h1("This is Heading One", hsl(120));
bil.insert.add(txt,0,10); //start at 0 end at 10;

const mega = TextTempl.megatron("This is Megatron", hsl(240));
bil.insert.add(mega,10,20); //start at 10 end at 30;

const footNote = TextTempl.footNote("This is footnote", hsl(180));
bil.insert.add(footNote,10,20); //start at 10 and end at 30

const ui = new Ui(bil);
bil.draw();
