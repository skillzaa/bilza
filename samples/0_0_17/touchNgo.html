import Bilza, { XAlignment, YAlignment, CompFactory as cf } from 
"./bilza_0_0_17.js";

let bil = new Bilza("bilza",75);
// bil.resizeCanvas(1000, 400);

let grid = cf.grid();
bil.insert.alwaysOn(grid);

let counter = cf.frameCounter();
counter.goto(0,0,20);
bil.insert.alwaysOn(counter);

let tst = cf.fillRect("red");
tst.duration = 120;

tst.goto(0,0,0, XAlignment.Left, YAlignment.Top);
//---Right Top
tst.animate(2,20,0,100,0,0,XAlignment.Left,
    XAlignment.Right);
//--Right Bottom    
tst.animate(22,40,100,100,0,100,XAlignment.Right,
    XAlignment.Right,YAlignment.Top,YAlignment.Bot);
//--Left Bottom    
tst.animate(42,60,100,0,100,100,XAlignment.Right,
    XAlignment.Left,YAlignment.Bot,YAlignment.Bot);
//--Right Top    
tst.animate(62,80,0,0,100,0,XAlignment.Left,
    XAlignment.Left,YAlignment.Bot,YAlignment.Bot);

bil.insert.append(tst, tst.duration);
////////////////////////////////////////////
bil.start();
