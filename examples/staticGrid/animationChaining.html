import Bilza, { Ui, CompFactory as cf, hsl} from "../../src/bilza.js";

import animationChaining from "./aniPresets/animationChaining.js";

let bil = new Bilza("bilza", 70);
// bil.resizeCanvas(500,300);

const grid = cf.grid("grey");
grid.showNumbers.set(false);
bil.insert.alwaysOn(grid);

const fill = cf.staticGrid(60,60,hsl(240));

fill.showBackground.set(true);
fill.colorBackground.set(hsl(120));
fill.showNumbers.set(true);

animationChaining(fill);
bil.insert.add(fill, 0, 60);
const ui = new Ui(bil);
bil.draw();
