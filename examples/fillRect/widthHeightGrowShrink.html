import Bilza, { Ui, CompFactory as cf, hsl} from "../../src/bilza.js";

let bil = new Bilza("bilza", 70);

const grid = cf.grid("grey");
bil.insert.alwaysOn(grid);

const comp = cf.fillRect(hsl(240));
comp.align(1,1);
comp.alignRotate(1,1);
comp.x.set(50);
comp.y.set(50);

comp.width.set(10);
comp.height.set(10);

comp.width.oscillate(1,60,10,80,5);
comp.height.oscillate(1,60,10,80,5);

bil.insert.add(comp, 0, 65);
const ui = new Ui(bil);
bil.draw();
