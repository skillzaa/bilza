Lines

Draw a shape with lines and then change the canvas size.

Demo

Code


    import Bilza from "../lib/Bilza_0_0_12.js";
    let bil = new Bilza("bilza", 600, 300, 10000);
    let g = bil.add.gridTempl.dashed();
    g.d.flagDrawNumbers = true;
    
    let lines = bil.add.lines(0, 10);
    lines.d.flagFilled = false;
    lines.d.responsive = true;
    lines.add(100, 10);
    lines.add(50, 100);
    bil.draw();
    
    setInterval(function () {
        bil.setCanvas(300, 150);
        bil.draw();
    }, 3000);
