import Bilza, { Ui, CompFactory as cf, hsl,getRandom} from "../../src/bilza.js";

let bil = new Bilza("bilza", 70);

bil.background.color.set(hsl(0,5,30));
bil.insert.alwaysOn(cf.grid());
//--------------------
const comp = cf.text("ABCDEFG",hsl(240));

comp.setxy(10,10);
comp.setPaddings(0);
//--font is 20 so the text will not conver up 80% of the screen
comp.fontSize.set(20);
//-- width is 80% butstill this will not change the size of the comp
// since that is dependent upon fontSize and we have not allowed
// that to be changed
comp.width.set(80);
//---Now we have allowed it to change fontSize such that the text 
//fits the comp.width.value(). 
//-- Keep in mind that the fontSize will be changed ONCE before 
// the first draw and after that its up to the user what he does with the text. 
//-- If you comment this line out the text will revert back
// to its initial value.
comp.fitToWidth.set(true);

comp.theme.color(0);
comp.showBackground.set(true);

bil.insert.add(comp,0,30);
//--------------------
const ui = new Ui(bil);
bil.draw();
