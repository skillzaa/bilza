let bil = new Bilza("bilza", 70);
bil.alwaysOn().frameCounter(0);
//--------------------
let flag = true;
//--------------------
const grid = bil.add(0,15).grid(10,10,hsl(240));

grid.setRespCellDims(flag);
grid.setxy(5,5);
grid.width.set(90);
grid.height.set(90);

//--------------------
const ui = new Ui(bil);
bil.draw();
//---------------------------
setInterval(function(){
    flag = !flag;
    grid.setRespCellDims(flag,20,20);
    bil.draw();
},2000);
