import Bilza, { Ui, CompFactory as cf, hsl,getRandom} from "../../src/bilza.js";

let bil = new Bilza("bilza", 70);

bil.background.color.set(hsl(240,100,15));
// bil.insert.alwaysOn(cf.grid(10,10,));
//--------------------
let startFrame = 0;
//--------------------
for (let row = 0; row < 100; row+=10) {
    for (let col = 0; col < 100; col+=20) {
        //....................................
        const comp = cf.circle(5,hsl(getRandom(0,360)));
        
        comp.setPaddings(0);

        comp.border.set(1);
        comp.colorBorder.set("red");
        comp.x.set(row);        
        comp.y.set(col);      
        // comp.width.set(10);
        // comp.height.set(10);
        bil.insert.add(comp,startFrame,120);
        startFrame++;  
        //....................................        
    }
}
//--------------------
const ui = new Ui(bil);
bil.draw();
