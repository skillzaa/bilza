import Bilza, { Ui,hsl} from "../../src/bilza.js";

let bil = new Bilza("bilza", 70);

bil.alwaysOn().grid(10,10,hsl(0,80,20));
bil.alwaysOn().frameCounter(0);
bil.background.color.set(hsl(240,30,10));

//=============================================
const title = bil.add(0,15).text("Arrow Pointer",hsl(120))
    .templ.hdg(1,0).entryAni.topIn().exitAni.fadeOut();

const target = bil.alwaysOn().circle(1,"orange");
target.setxy(90,90);
target.color.goto(10,"green");

const arrow = bil.add(0,15).arrow(10,10,20,20,hsl(0));
arrow.headFilled.set(false);

arrow.x2.animate(2,10,20,90);
arrow.y2.animate(2,10,20,90);
//-------------------
const ui = new Ui(bil);
bil.draw();
