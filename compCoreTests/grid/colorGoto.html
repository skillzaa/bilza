function getText(content,y = 50){
    const caption = cf.text("",hsl(240));
    caption.content.set(content);
    caption.align(1,1)// --x align mid , y align bottom
    caption.x.set(50);
    caption.y.set(y);
    caption.fontSize.set(35);
    bil.insert.add(caption,0,15);
}
//--------------------
const ui = new Ui(bil);
bil.draw();
